  <script>

    const scene = document.getElementById('scene');

    const cardWrap = document.getElementById('cardWrap');

    const hint = document.getElementById('hint');

    let opened = false;


    function createConfetti(count = 110) {

      const colors = ['#13c8c8', '#74efe6', '#a6fff7', '#7dd3fc', '#fde68a', '#ffffff'];

      for (let i = 0; i < count; i++) {

        const piece = document.createElement('div');

        piece.className = 'confetti';

        piece.style.background = colors[Math.floor(Math.random() * colors.length)];

        piece.style.left = (50 + (Math.random() * 12 - 6)) + '%';

        piece.style.top = (44 + (Math.random() * 10 - 5)) + '%';

        piece.style.setProperty('--x', `${(Math.random() * 520 - 260).toFixed(0)}px`);

        piece.style.setProperty('--y', `${(Math.random() * 360 - 160).toFixed(0)}px`);

        piece.style.setProperty('--r', `${(Math.random() * 720 - 360).toFixed(0)}deg`);

        piece.style.animation = `burst ${2.8 + Math.random() * 1.8}s cubic-bezier(.2,.8,.2,1) forwards`;

        scene.appendChild(piece);

        setTimeout(() => piece.remove(), 2300);

      }

    }


    function openCard() {

      if (opened) return;

      opened = true;

      cardWrap.classList.add('open');

      scene.classList.add('open');

      hint.style.opacity = '0';

      setTimeout(() => hint.remove(), 400);

      setTimeout(() => createConfetti(54), 300);

    }


    cardWrap.addEventListener('click', openCard);

    cardWrap.addEventListener('keydown', (e) => {

      if (e.key === 'Enter' || e.key === ' ') {

        e.preventDefault();

        openCard();

      }

    });
